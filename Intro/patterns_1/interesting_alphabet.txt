import java.util.Scanner;
public class Solution {
	public static void main(String[] args) {
		//Your code goes here
        Scanner s = new Scanner(System.in);
        int N = s.nextInt();
        int i = N;
        while(i >= 1 && N >= 1 && N <= 26){
            int j = i;
            while(j <= N){
                System.out.print((char)('A' + j - 1));
                j++;
            }
            System.out.println();
            i--;
        }
	}
}
