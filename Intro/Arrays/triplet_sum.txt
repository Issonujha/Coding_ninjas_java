
public class Solution {
    
    public static int findTriplet(int[] arr, int x) {
    	//Your code goes here
        int count=0;
        for(int i=0; i<arr.length;i++) {
            for(int j=0; j<arr.length; j++) {
                for(int k=0; k<arr.length;k++) {
                    if((arr[i]+arr[j]+arr[k]==x) && i!=j && i!=k && j!=k ) {
                        count++;
                    }
                }
            }
        }
        count = count/6;
        return count;
    }

}
