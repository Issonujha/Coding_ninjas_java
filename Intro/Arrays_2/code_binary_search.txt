
public class Solution { 

    public static int binarySearch(int[] arr, int x) {
    
        int start = 0;
        int end = arr.length - 1;
        int mid = 0; 
        while(start<=end){
            
            mid = (start+end)/2;
        	if(arr[mid]==x) {
                return mid;
            }
            else if(x<arr[mid]) {
                end = mid-1;
            }
            else {
                start = mid + 1;
            }
        }
        return -1;
    }
}
