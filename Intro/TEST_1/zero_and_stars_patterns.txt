import java.util.Scanner;
public class Solution {

	public static void main(String[] args) {

		/*  Read input as specified in the question.
    		 * Print output as specified in the question.
    		 */
        Scanner s = new Scanner(System.in);
        int n = s.nextInt();
        int i, j;
        for(i=1;i<=n;i++)
        {
        	for(j=1;j<=n;j++)
            {
                if(i==j)
                {
                    System.out.print("*");
                }
                else{
                    System.out.print("0");
                }
                
            }
            System.out.print("*");
            for(j=(n);j>=1;j--){
                if(i==j)
                {
                    System.out.print("*");
                }
                else
                {
                    System.out.print("0");
                }
            }
        	System.out.println();
        }
		
	}	


}
