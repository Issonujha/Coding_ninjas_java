
public class Solution {

	public static void rowWiseSum(int[][] mat) {
		//Your code goes here
        int rows = mat.length;
        int col = 0;
        if(rows>=1) {
        	col = mat[0].length;
        }
        for(int i=0; i<rows; i++) {
            int sum = 0;
            for(int j=0; j<col; j++) {
                sum = sum + mat[i][j];
            }
            System.out.print(sum + " ");
        }
	}
}
