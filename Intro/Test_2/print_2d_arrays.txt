public class solution {
	public static void print2DArray(int input[][]) {
		// Write your code here
        int row = input.length;
        int col = 0;
        if(row>0) {
            col = input[0].length;
        }
        int n = row;
		for(int i=0; i<row;i++) {
            int count = 0;
            for(int j=0; j<col && count<=n-i;j++) {
                System.out.print(input[i][j]+ " ");
                if(j==col-1) {
                    count++;
                    if(count == n-i) {
                        break;
                    }
                    else {
                        j=-1;
                        System.out.println();
                    }
                }
            }
            System.out.println();
        }
	}
}
