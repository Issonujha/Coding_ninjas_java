import java.util.Scanner;
public class Main {
	
	public static void main(String[] args) {
		// Write your code here
		Scanner s = new Scanner(System.in);
        int N = s.nextInt();
        int total = 0;
        int j = 1;
        while(N!=0){
            int rem = N%10;
        	N = N/10;
            total = total + rem*j;
            j *=2;
        }
        
        System.out.print(total);
	}
}
