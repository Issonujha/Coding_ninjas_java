    public static String replace(String input, int i, String s) {
        String n = "";
        if(i>input.length()-2) {
            if(i==input.length()-1){
            	n = n + input.charAt(i);
            }
            return n;
        }
        if(input.charAt(i)=='p' && input.charAt(i+1)=='i') {
            s = "3.14" + replace(input, i+2, s);
        }
        else {
        	s = input.charAt(i) + replace(input, i+1, s);
        }
        return s;
    }
