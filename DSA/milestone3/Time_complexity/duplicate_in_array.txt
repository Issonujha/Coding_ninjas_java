public class Solution {

	public static int findDuplicate(int[] arr) {
		//Your code goes here
        int sum = 0;
        int n = arr.length-2;
        for(int i = 0; i<arr.length; i++) {
            sum += arr[i];
        }
        int a = n*(n+1)/2;
        int pair = sum-a;
        return pair;
	}
}
