public class Solution {

    public static int arrayRotateCheck(int[] arr){
    	//Your code goes here
        int count = 0;
        for(int i = 0; i<arr.length-1; i++) {
            if(arr[i]<arr[i+1]) {
                count++;
            }
            else {
                break;
            }
        }
        if(count>0) {
        	count = count+1;
        }
        if(count==arr.length) {
            count = 0;
        }
        return count;
    }

}
