public class Solution {

	public static int findDuplicate(int[] arr) {
		//Your code goes here
        int sum = 0;
        for(int i = 0; i<arr.length; i++) {
            sum+=arr[i];
        }
        int sum2 = 0;
        for(int i = 0; i<arr.length-1; i++) {
            sum2 += i;
        }
        int dup = sum-sum2;
        
        return dup;
	}
}
