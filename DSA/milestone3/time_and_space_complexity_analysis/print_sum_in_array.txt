public class Solution {	

	public static int pairSum(int[] arr, int num) {
		//Your code goes here
        int count = 0;
		for(int i = 0; i<arr.length; i++) {
		for(int j= 0; j<arr.length; j++) {
		int sum = 0;
		if(i==j) {
		continue;
		}
		sum = arr[i] + arr[j];
		if(sum==num) {
		count++;
		}
		}
		}
		count = count/2;
		return count;
	}
}
