    public static int minCount(int n, int[]output) {
        for(int i = 4; i<=n; i++) {
            output[i] = i;
            
            
            for(int k = 1; k<=Math.ceil(Math.sqrt(i)); k++) {
                int temp = k*k;
                if(temp>i) {
                    break;
                }
                else {
                    output[i] = Math.min(output[i], 1 + output[i-temp]);
                }
            }
        }
        return output[n];
    }
